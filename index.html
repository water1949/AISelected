2.3 Python SDK集成

python 
import ollama
from typing import List, Dict

class LocalAgent:
    def __init__(self, model_name: str = "my-agent"):
        self.model = model_name
        self.client = ollama.Client(host='http://localhost:11434')
        self.memory: List[Dict] = []
    
    def chat(self, message: str, tools: List[Dict] = None) -> str:
        """支持工具调用的对话接口"""
        messages = self.memory + [{"role": "user", "content": message}]
        
        response = self.client.chat(
            model=self.model,
            messages=messages,
            tools=tools,
            options={
                "temperature": 0.7,
                "num_predict": 2048  # 限制输出长度避免幻觉
            }
        )
        
        # 更新记忆（滑动窗口管理）
        self.memory.append({"role": "user", "content": message})
        self.memory.append({
            "role": "assistant", 
            "content": response['message']['content']
        })
        
        # 保持最近10轮对话
        if len(self.memory) > 20:
            self.memory = self.memory[-20:]
            
        return response['message']['content']
    
    def clear_memory(self):
        self.memory = []

# 使用示例
agent = LocalAgent()
response = agent.chat("帮我规划一个3天2晚的北京旅行行程")
print(response)

5.2 自定义工具注册框架

python 
import inspect
from typing import Callable, Dict, Any
import json

class ToolRegistry:
    """工具注册中心"""
    def __init__(self):
        self._tools: Dict[str, Dict] = {}
    
    def register(self, func: Callable = None, name: str = None, description: str = None):
        """装饰器注册工具"""
        def decorator(f):
            tool_name = name or f.__name__
            tool_desc = description or f.__doc__ or "No description"
            
            # 解析函数签名
            sig = inspect.signature(f)
            params = {}
            for param_name, param in sig.parameters.items():
                param_info = {"type": "string"}  # 默认字符串
                if param.default != inspect.Parameter.empty:
                    param_info["default"] = param.default
                params[param_name] = param_info
            
            self._tools[tool_name] = {
                "function": f,
                "schema": {
                    "name": tool_name,
                    "description": tool_desc,
                    "parameters": {
                        "type": "object",
                        "properties": params,
                        "required": list(params.keys())
                    }
                }
            }
            return f
        
        if func:
            return decorator(func)
        return decorator
    
    def get_schemas(self) -> List[Dict]:
        """获取OpenAI格式的工具描述"""
        return [{"type": "function", "function": t["schema"]} for t in self._tools.values()]
    
    def execute(self, tool_name: str, arguments: Dict) -> Any:
        """执行工具"""
        if tool_name not in self._tools:
            raise ValueError(f"未知工具: {tool_name}")
        
        func = self._tools[tool_name]["function"]
        return func(**arguments)

# 实例化注册中心
registry = ToolRegistry()

# 注册示例工具
@registry.register(description="查询数据库")
def query_database(sql: str, limit: int = 10):
    """模拟数据库查询"""
    return f"执行SQL: {sql}, 返回{limit}条结果..."

@registry.register(description="发送通知")
def send_notification(user_id: str, message: str, level: str = "info"):
    """发送系统通知"""
    return f"已发送{level}级别消息给{user_id}: {message}"

# Agent使用
class FunctionCallAgent:
    def __init__(self, llm_client, tool_registry):
        self.client = llm_client
        self.tools = tool_registry
    
    def run(self, user_input: str) -> str:
        # 第一次调用：获取工具调用决策
        response = self.client.chat.completions.create(
            model="Qwen2.5-14B-Instruct",
            messages=[{"role": "user", "content": user_input}],
            tools=self.tools.get_schemas(),
            tool_choice="auto"
        )
        
        msg = response.choices[0].message
        
        # 处理工具调用
        if msg.tool_calls:
            results = []
            for tool_call in msg.tool_calls:
                func_name = tool_call.function.name
                args = json.loads(tool_call.function.arguments)
                
                # 执行工具
                result = self.tools.execute(func_name, args)
                results.append({
                    "tool_call_id": tool_call.id,
                    "role": "tool",
                    "name": func_name,
                    "content": str(result)
                })
            
            # 第二次调用：生成最终回复
            final_response = self.client.chat.completions.create(
                model="Qwen2.5-14B-Instruct",
                messages=[
                    {"role": "user", "content": user_input},
                    msg.model_dump(),
                    *results
                ]
            )
            return final_response.choices[0].message.content
        
        return msg.content

# 使用
agent = FunctionCallAgent(client, registry)
result = agent.run("查询最近注册用户，并发送欢迎消息给user_123")
print(result)

6.2 完整RAG Agent流程

python 
class RAGAgent:
    def __init__(self, llm_base_url: str, embedding_model_path: str):
        self.client = OpenAI(base_url=llm_base_url, api_key="not-needed")
        self.embedder = SentenceTransformer(embedding_model_path)
        self.index = None
        self.documents = []
        
    def build_knowledge_base(self, docs: List[str]):
        """构建知识库"""
        self.documents = docs
        embeddings = self.embedder.encode(docs, normalize_embeddings=True)
        
        dimension = embeddings.shape[1]
        self.index = faiss.IndexFlatIP(dimension)
        self.index.add(embeddings)
        
    def retrieve(self, query: str, top_k: int = 3) -> List[str]:
        """检索相关文档"""
        query_vec = self.embedder.encode([query], normalize_embeddings=True)
        scores, indices = self.index.search(query_vec, top_k)
        
        return [self.documents[i] for i in indices[0] if scores[0][i] > 0.5]
    
    def chat(self, query: str) -> str:
        """RAG对话"""
        # 检索上下文
        contexts = self.retrieve(query)
        context_str = "\n".join([f"[{i+1}] {ctx}" for i, ctx in enumerate(contexts)])
        
        # 构造Prompt
        prompt = f"""基于以下参考资料回答问题。如果资料不足以回答，请说明"根据现有资料无法确定"。

参考资料：
{context_str}

用户问题：{query}

请给出准确、简洁的回答："""
        
        response = self.client.chat.completions.create(
            model="Qwen2.5-14B-Instruct",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.3  # RAG需要低温度确保准确性
        )
        
        return response.choices[0].message.content

七、多Agent系统架构

7.1 多Agent协作框架（Controller模式）

python 
from enum import Enum
from typing import List, Optional
import concurrent.futures

class AgentRole(Enum):
    PLANNER = "planner"      # 任务规划
    RESEARCHER = "researcher" # 信息搜集
    CODER = "coder"          # 代码生成
    REVIEWER = "reviewer"    # 结果审查

class SubAgent:
    """专业子Agent"""
    def __init__(self, role: AgentRole, model_endpoint: str, system_prompt: str):
        self.role = role
        self.client = OpenAI(base_url=model_endpoint, api_key="not-needed")
        self.system_prompt = system_prompt
        
    def execute(self, task: str, context: List[Dict] = None) -> str:
        messages = [{"role": "system", "content": self.system_prompt}]
        if context:
            messages.extend(context)
        messages.append({"role": "user", "content": task})
        
        response = self.client.chat.completions.create(
            model="Qwen2.5-14B-Instruct",
            messages=messages,
            temperature=0.7
        )
        return response.choices[0].message.content

class MultiAgentSystem:
    """多Agent控制器"""
    def __init__(self):
        self.agents: Dict[AgentRole, SubAgent] = {}
        self.endpoint = "http://localhost:8000/v1"
        
    def register_agent(self, role: AgentRole, system_prompt: str):
        """注册Agent"""
        self.agents[role] = SubAgent(role, self.endpoint, system_prompt)
        
    def run_workflow(self, user_request: str) -> Dict:
        """执行多Agent工作流"""
        
        # Step 1: Planner分解任务
        planner = self.agents[AgentRole.PLANNER]
        plan = planner.execute(
            f"将以下请求分解为可执行步骤（JSON格式）：{user_request}"
        )
        
        # Step 2: 并行执行子任务
        results = {}
        with concurrent.futures.ThreadPoolExecutor() as executor:
            futures = {}
            
            if AgentRole.RESEARCHER in self.agents and "搜索" in plan:
                futures[executor.submit(
                    self.agents[AgentRole.RESEARCHER].execute, 
                    "搜集相关信息", 
                    [{"role": "assistant", "content": plan}]
                )] = "research"
                
            if AgentRole.CODER in self.agents and "代码" in plan:
                futures[executor.submit(
                    self.agents[AgentRole.CODER].execute,
                    "编写代码",
                    [{"role": "assistant", "content": plan}]
                )] = "code"
            
            for future in concurrent.futures.as_completed(futures):
                task_name = futures[future]
                try:
                    results[task_name] = future.result()
                except Exception as e:
                    results[task_name] = f"Error: {str(e)}"
        
        # Step 3: Reviewer整合结果
        if AgentRole.REVIEWER in self.agents:
            final_output = self.agents[AgentRole.REVIEWER].execute(
                f"整合以下结果并生成最终回复：\n{json.dumps(results, ensure_ascii=False)}",
                [{"role": "assistant", "content": plan}]
            )
        else:
            final_output = json.dumps(results, ensure_ascii=False)
            
        return {
            "plan": plan,
            "intermediate_results": results,
            "final_output": final_output
        }

# 使用示例
system = MultiAgentSystem()

# 注册各角色Agent

import time
from contextlib import contextmanager
from collections import deque
import psutil
import GPUtil

class AgentMonitor:
    def __init__(self):
        self.latency_history = deque(maxlen=100)  # 延迟历史
        self.token_throughput = []  # 吞吐量
        self.error_count = 0
        
    @contextmanager
    def track_request(self):
        """跟踪单次请求性能"""
        start_time = time.time()
        start_tokens = self._get_total_tokens()
        
        try:
            yield self
            # 记录成功指标
            latency = time.time() - start_time
            self.latency_history.append(latency)
            
        except Exception as e:
            self.error_count += 1
            raise e
            
    def _get_total_tokens(self):
        # 实际应从vLLM metrics接口获取
        return 0
    
    def get_stats(self):
        """获取统计报告"""
        if not self.latency_history:
            return "暂无数据"
            
        avg_latency = sum(self.latency_history) / len(self.latency_history)
        p95_latency = sorted(self.latency_history)[int(len(self.latency_history)*0.95)]
        
        # 系统资源
        cpu_percent = psutil.cpu_percent()
        memory = psutil.virtual_memory()
        gpus = GPUtil.getGPUs()
        gpu_load = gpus[0].load * 100 if gpus else 0
        
        return {
            "avg_latency_sec": round(avg_latency, 2),
            "p95_latency_sec": round(p95_latency, 2),
            "error_rate": f"{(self.error_count / len(self.latency_history) * 100):.2f}%",
            "cpu_percent": cpu_percent,
            "memory_percent": memory.percent,
            "gpu_load_percent": round(gpu_load, 1)
        }

# diagnose.py - 运行环境诊断
import torch
import subprocess
import os

def check_environment():
    print("=== 本地模型部署环境诊断 ===\n")
    
    # 1. CUDA检查
    print("1. CUDA可用性:")
    print(f"   PyTorch版本: {torch.__version__}")
    print(f"   CUDA可用: {torch.cuda.is_available()}")
    if torch.cuda.is_available():
        print(f"   CUDA版本: {torch.version.cuda}")
        print(f"   GPU数量: {torch.cuda.device_count()}")
        for i in range(torch.cuda.device_count()):
            print(f"   GPU {i}: {torch.cuda.get_device_name(i)}")
            print(f"   显存: {torch.cuda.get_device_properties(i).total_memory / 1024**3:.2f} GB")
    
    # 2. 显存碎片检查
    print("\n2. 显存状态:")
    if torch.cuda.is_available():
        for i in range(torch.cuda.device_count()):
            torch.cuda.set_device(i)
            allocated = torch.cuda.memory_allocated() / 1024**3
            reserved = torch.cuda.memory_reserved() / 1024**3
            print(f"   GPU {i}: 已分配 {allocated:.2f}GB, 预留 {reserved:.2f}GB")
    
    # 3. 端口检查
    print("\n3. 常用端口占用:")
    ports = [11434, 8000, 8080, 3000]
    for port in ports:
        result = subprocess.run(
            ["lsof", "-i", f":{port}"], 
            capture_output=True, 
            text=True
        )
        status = "已被占用" if result.returncode == 0 else "空闲"
        print(f"   端口 {port}: {status}")
    
    # 4. 磁盘空间
    print("\n4. 磁盘空间:")
    stat = os.statvfs("/")
    free_gb = (stat.f_bavail * stat.f_frsize) / 1024**3
    print(f"   根目录剩余: {free_gb:.2f} GB")
    
    print("\n=== 诊断完成 ===")

if __name__ == "__main__":
    check_environment()
